ules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // PERMISSÃO PÚBLICA PARA O CAMINHO DOS RELATÓRIOS
    // Seu aplicativo salva os dados no caminho /artifacts/{appId}/public/data/actions
    // Esta regra permite que qualquer usuário autenticado (incluindo anônimos)
    // leia e escreva dados neste caminho público.
    match /artifacts/{appId}/public/data/{collection}/{document} {
      // Permite leitura e escrita se o usuário estiver autenticado
      // (request.auth != null é verdadeiro para autenticação anônima ou token customizado).
      allow read, write: if request.auth != null;
    }

    // Regra padrão de proteção de dados privados do usuário (apenas para referência)
    match /artifacts/{appId}/users/{userId}/{documents=**} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
  }
}
